105  private final void failIfNotOpen() throws NamingException {
106    final int state = this.state;
107    if (state > 0) {
108      throw (NamingException)new NamingException("context is closing or closed").initCause(new IllegalStateException("context is closing or closed"));
109    }
110  }

112  private final void failIfClosed() throws NamingException {
113    final int state = this.state;
114    if (state >= CLOSED) {
115      throw (NamingException)new NamingException("context is closing or closed").initCause(new IllegalStateException("context is closing or closed"));
116    }
117  }

119  protected final Name toCompoundName(final Name name) throws NamingException {
120    Objects.requireNonNull(name);
121    final Name returnValue;
122    if (name instanceof CompositeName) {
123      returnValue = this.toCompoundName((CompositeName)name);
124    } else {
125      failIfNotOpen();
126      returnValue = name;
127    }
128    return returnValue;
129  }

131  protected final Name toCompoundName(final CompositeName compositeName) throws NamingException {
132    Objects.requireNonNull(compositeName);
133    failIfNotOpen();
134    final NameParser nameParser = this.getNameParser(EMPTY_NAME);
135    if (nameParser == null) {
136      throw (NamingException)new NamingException().initCause(new IllegalStateException("getNameParser(\"\") == null"));
137    }
138    final Name returnValue = nameParser.parse("");
139    assert returnValue != null;
140    assert returnValue.size() == 0;
141    assert returnValue.isEmpty();
142    final int size = compositeName.size();
143    assert size >= 0;
144    if (size > 0) {
145      int i = 0;
146      for (; i < size; i++) {
147        final String compositeNameComponent = compositeName.get(i);
148        assert compositeNameComponent != null;
149        if (!compositeNameComponent.isEmpty()) {
150          String compositeNameNonUrlComponent = compositeNameComponent;
151          if (i == 0) {
152            final Matcher urlMatcher = urlPattern.matcher(compositeNameComponent);
153            assert urlMatcher != null;
154            if (urlMatcher.matches()) {
155              compositeNameNonUrlComponent = urlMatcher.group(1);
156            }
157          }
158          assert compositeNameNonUrlComponent != null;
159          if (!compositeNameNonUrlComponent.isEmpty()) {
160            try {
161              returnValue.addAll(nameParser.parse(compositeNameNonUrlComponent));
162            } catch (final InvalidNameException parsingFailed) {
163              break;
164            }
165          }
166        }
167      }
168
169      if (i < size) {
170        // at this point i identifies the suffix to the CannotProceedException
171        // TODO: deal with federation
172        throw new InvalidNameException("compositeName could not be entirely parsed; successful parse: " + returnValue + "; remaining: " + compositeName.getSuffix(i));
173      }
174    }
175    return returnValue;
176  }

182  protected final Object get(final K key, final Name name) throws NamingException {
183    failIfClosed();
184    final Object returnValue;
185    Object temp = null;
186    try {
187      temp = NamingManager.getObjectInstance(this.get(key), name, this, this.environment);
188    } catch (final RuntimeException throwMe) {
189      throw throwMe;
190    } catch (final NamingException throwMe) {
191      throw throwMe;
192    } catch (final Exception otherStuff) {
193      throw (NamingException)new NamingException(otherStuff.getMessage()).initCause(otherStuff);
194    } finally {
195      returnValue = temp;
196    }
197    return returnValue;
198  }

214  @Override
215  public NamingEnumeration<NameClassPair> list(final Name name) throws NamingException {
216    failIfNotOpen();
217    Objects.requireNonNull(name);
218    final NamingEnumeration<NameClassPair> returnValue;
219    if (name.isEmpty()) {
220      returnValue = new NameClassPairEnumeration<K>(this.keySet()) {
221          @Override
222          protected final Object get(final K key) throws NamingException {
223            final NameParser nameParser = getNameParser(EMPTY_NAME);
224            if (nameParser == null) {
225              return AbstractContext.this.get(key, null);
226            } else {
227              return AbstractContext.this.get(key, nameParser.parse(key.toString()));
228            }
229          }
230        };
231    } else {
232      final Object target = this.lookup(name);
233      if (target instanceof Context) {
234        final Context context = (Context)target;
235        NamingEnumeration<NameClassPair> temp = null;
236        try {
237          temp = context.list(EMPTY_NAME);
238        } finally {
239          returnValue = temp;
240        }
241      } else {
242        returnValue = null;
243        throw new NotContextException(name + " cannot be listed");
244      }
245    }
246    return returnValue;
247  }

251    failIfClosed(); // ...but not CLOSING
252    Objects.requireNonNull(name);
253    final NamingEnumeration<Binding> returnValue;
254    if (name.isEmpty()) {
255      returnValue = new BindingEnumeration<K>(this.keySet()) {
256          @Override
257          protected final Object get(final K key) throws NamingException {
258            final NameParser nameParser = getNameParser(EMPTY_NAME);
259            if (nameParser == null) {
260              return AbstractContext.this.get(key, null);
261            } else {
262              return AbstractContext.this.get(key, nameParser.parse(key.toString()));
263            }
264          }
265        };
266    } else {
267      final Object target = this.lookup(name);
268      if (target instanceof Context) {
269        final Context context = (Context)target;
270        NamingEnumeration<Binding> temp = null;
271        try {
272          temp = context.listBindings(EMPTY_NAME);
273        } finally {
274          returnValue = temp;
275        }
276      } else {
277        returnValue = null;
278        throw new NotContextException(name + " cannot be listed");
279      }
280    }
281    return returnValue;
282  }

289  @Override
290  public Object lookup(final Name name) throws NamingException {
291    failIfNotOpen();
292    Objects.requireNonNull(name);
293    
294    final Name compoundName = toCompoundName(name);
295    assert compoundName != null;
296
297    final Object returnValue;
298
299    final int size = compoundName.size();
300    if (size == 0) {
301      assert compoundName.isEmpty();
302      returnValue = this.copy();
303    } else {
304      assert size > 0;
305      final K mapKey = this.extractKey(compoundName);
306      if (mapKey == null ||
307          (mapKey instanceof String && ((String)mapKey).isEmpty()) ||
308          (mapKey instanceof Name && ((Name)mapKey).isEmpty())) {
309        throw new InvalidNameException(name.toString());
310      }
311      final Object value = this.get(mapKey, compoundName);
312      if (size == 1) {
313        returnValue = value;
314        if (returnValue == null) {
315          throw new NameNotFoundException(name.toString());
316        }
317      } else if (value instanceof Context) {
318        assert size > 1;
319        returnValue = ((Context)value).lookup(compoundName.getSuffix(1));
320      } else {
321        throw new NotContextException(mapKey.toString());
322      }
323    }
324    
325    return returnValue;
326  }

334  protected void bind(final Name name, final Object obj, final boolean rebindPermitted) throws NamingException {
335    Objects.requireNonNull(name);
336    if (name.isEmpty()) {
337      throw new InvalidNameException("name.isEmpty()");
338    }
339
340    final Name compoundName = this.toCompoundName(name);
341    assert compoundName != null;
342    if (compoundName.isEmpty()) {
343      throw new InvalidNameException(name.toString());
344    }
345    
346    final int size = compoundName.size();
347    assert size > 0;
348
349    final K mapKey = this.extractKey(compoundName);
350    if (mapKey == null ||
351        (mapKey instanceof String && ((String)mapKey).isEmpty()) ||
352        (mapKey instanceof Name && ((Name)mapKey).isEmpty())) {
353      throw new InvalidNameException(name.toString());
354    }
355    
356    final Object value = this.get(mapKey, compoundName);
357    if (size == 1) {
358      if (value != null) {
359        if (rebindPermitted) {
360          final Object old = this.remove(mapKey);
361          assert old == value;
362        } else {
363          throw new NameAlreadyBoundException(name.toString());
364        }
365      }
366      this.put(mapKey, NamingManager.getStateToBind(obj, this.getNameParser(EMPTY_NAME).parse(mapKey.toString()), this, this.environment));
367    } else if (value instanceof Context) {
368      assert size > 1;
369      ((Context)value).bind(compoundName.getSuffix(1), obj);
370    } else {
371      assert size > 1;
372      throw new NotContextException(mapKey.toString());
373    }
374  }

395  public void unbind(final Name name) throws NamingException {
396    failIfNotOpen();
397    Objects.requireNonNull(name);
398    if (name.isEmpty()) {
399      throw new InvalidNameException("name.isEmpty()");
400    }
401
402    final Name compoundName = this.toCompoundName(name);
403    assert compoundName != null;
404    if (compoundName.isEmpty()) {
405      throw new InvalidNameException(name.toString());
406    }
407    
408    final int size = compoundName.size();
409    assert size > 0;
410
411    final K mapKey = this.extractKey(compoundName);
412    if (mapKey == null ||
413        (mapKey instanceof String && ((String)mapKey).isEmpty()) ||
414        (mapKey instanceof Name && ((Name)mapKey).isEmpty())) {
415      throw new InvalidNameException(name.toString());
416    }
417
418    if (size == 1) {
419      this.remove(mapKey);
420    } else {
421      assert size > 1;
422      final Object value = this.get(mapKey, compoundName);
423      if (value instanceof Context) {
424        ((Context)value).unbind(compoundName.getSuffix(1));
425      } else {
426        throw new NotContextException(mapKey.toString());
427      }
428    }

465  @Override
466  public void destroySubcontext(final Name name) throws NamingException {
467    failIfNotOpen();
468    Objects.requireNonNull(name);
469    if (name.isEmpty()) {
470      throw new InvalidNameException("name.isEmpty()");
471    }
472    final Name compoundName = this.toCompoundName(name);
473    assert compoundName != null;
474    if (!compoundName.isEmpty()) {
475      final K mapKey = this.extractKey(compoundName);
476      if (mapKey == null ||
477          (mapKey instanceof String && ((String)mapKey).isEmpty()) ||
478          (mapKey instanceof Name && ((Name)mapKey).isEmpty())) {
479        throw new InvalidNameException(name.toString());
480      }
481      final int size = compoundName.size();
482      assert size > 0;
483      Object value = this.get(mapKey, compoundName);
484      if (size == 1) {
485        if (value != null) {
486          if (value instanceof Context) {
487            final Context subcontext = (Context)value;
488            final Enumeration<?> enumeration = subcontext.list(EMPTY_NAME);
489            if (enumeration != null && enumeration.hasMoreElements()) {
490              throw new ContextNotEmptyException(name.toString());
491            }
492            subcontext.close();
493            final Object old = this.remove(mapKey);
494            assert old == subcontext;
495          } else {
496            throw new NotContextException(mapKey + " in " + name.toString());
497          }
498        }
499      } else if (value instanceof Context) {
500        assert size > 1;
501        ((Context)value).destroySubcontext(compoundName.getSuffix(1));
502      } else {
503        throw new NotContextException(mapKey + " in " + name.toString());
504      }
505    } else {
506      // Specification says that this method should be a no-op if the
507      // compound name is not bound in this context; an empty compound
508      // name *is* this context so it is not bound.
509    }
510  }

517  @Override
518  public Context createSubcontext(final Name name) throws NamingException {
519    failIfNotOpen();
520    Objects.requireNonNull(name);
521    if (name.isEmpty()) {
522      throw new InvalidNameException("name.isEmpty()");
523    }
524    final Name compoundName = toCompoundName(name);
525    assert compoundName != null;
526    final int size = compoundName.size();
527    assert size > 0;
528    final K mapKey = this.extractKey(compoundName);
529    if (mapKey == null ||
530        (mapKey instanceof String && ((String)mapKey).isEmpty()) ||
531        (mapKey instanceof Name && ((Name)mapKey).isEmpty())) {
532      throw new InvalidNameException(name.toString());
533    }
534    if (this.containsKey(mapKey)) {
535      throw new NameAlreadyBoundException(name.toString());
536    }
537    final Object value = this.get(mapKey, compoundName);
538    final Context returnValue;
539    if (size == 1) {
540      returnValue = this.newContext(name); // yes, name; not compoundName
541      if (returnValue == null) {
542        throw (NamingException)new NamingException().initCause(new IllegalStateException("newContext() == null"));
543      }
544      final Object old = this.put(mapKey, returnValue);
545      assert old == null;
546    } else if (value instanceof Context) {
547      assert size > 1;
548      returnValue = ((Context)value).createSubcontext(compoundName.getSuffix(1));
549    } else {
550      throw new NotContextException(mapKey + " in " + name.toString());
551    }
552    return returnValue;
553  }

571  @Override
572  public NameParser getNameParser(final Name name) throws NamingException {
573    failIfClosed();
574    // The word "NameParser" shows up only twice, total, in what
575    // passes for the JNDI specification
576    // (https://docs.oracle.com/javase/8/docs/technotes/guides/jndi/spec/jndi/).
577    // There is no indication what the name parameter is used for in
578    // this method.
579    //
580    // The javadoc says "Retrieves the parser associated with the
581    // named context."  This suggests that the supplied Name must
582    // (ultimately) name a Context, but what if the supplied Name is a
583    // CompositeName, as it will be when getNameParser(String) is
584    // called?
585    //
586    // The tutorial
587    // (https://docs.oracle.com/javase/jndi/tutorial/provider/basics/parser.html)
588    // also implies that the name supplied here must be a compound
589    // name, and an example does a lookup(Name) on it to verify it
590    // exists.  But this is circular: part of what lookup(Name) must
591    // do is parse components of what might be a CompositeName, and to
592    // do that it needs a name parser.
593    //
594    // All of this craziness suggests that this method is used to get
595    // the NameParser affiliated with a Context implementation stored
596    // in THIS Context implementation under the supplied compound
597    // name.  Recall that the supplied compound name might be (and
598    // usually is) empty, which means this very context itself.
599    //
600    // If we are handed a CompositeName, it's hard to know what to do,
601    // since name parsing is so very fundamental to separating out the
602    // concepts of composite and compound names.  For example, if we
603    // were to try to use the toCompoundName(Name) method here, we
604    // would get an infinite loop, because it uses a NameParser to
605    // parse the CompositeName components.
606    final NameParser returnValue;
607    if (name == null || name.isEmpty()) {
608      returnValue = this.nameParser;
609    } else {
610      final Object potentialSubcontext = this.lookup(name);
611      if (potentialSubcontext instanceof Context) {
612        returnValue = ((Context)potentialSubcontext).getNameParser(EMPTY_NAME);
613      } else {
614        throw new NotContextException(name.toString());
615      }
616    }
617    return returnValue;
618  }

660  @Override
661  public void close() throws NamingException {
662    final int state = this.state;
663    if (state == 0) {
664      this.state = CLOSING;
665    }
666    final NamingEnumeration<Binding> contents = this.listBindings(EMPTY_NAME);
667    if (contents != null) {
668      while (contents.hasMore()) {
669        final Binding binding = contents.next();
670        if (binding != null) {
671          final Object value = binding.getObject();
672          if (value instanceof Context) {
673            ((Context)value).close();
674          }
675        }
676      }
677    }
678    this.state = CLOSED;
679  }
