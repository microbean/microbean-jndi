017package org.microbean.jndi;
018
019import java.util.HashMap;
020import java.util.Hashtable;
021import java.util.Map;
022import java.util.Objects;
023import java.util.Set;
024
025import javax.naming.Context;
026import javax.naming.InvalidNameException;
027import javax.naming.Name;
028import javax.naming.NameAlreadyBoundException;
029import javax.naming.NameNotFoundException;
030import javax.naming.NameParser;
031import javax.naming.NamingException;
032
033public class MapContext extends AbstractContext<String> {
034
035  private final Map<String, Object> map;
036
037  public MapContext(final NameParser nameParser) {
038    this(null, null, nameParser, null);
039  }
040  
041  public MapContext(final Hashtable<?, ?> environment, final NameParser nameParser) {
042    this(null, environment, nameParser, null);
043  }
044  
045  public MapContext(final Map<? extends String, ?> map, final Hashtable<?, ?> environment, final NameParser nameParser, final Name prefix) {
046    super(environment, nameParser, prefix);
047    if (map == null || map.isEmpty()) {
048      this.map = new HashMap<>();
049    } else {
050      this.map = new HashMap<>(map);
051    }
052  }
053
054  @Override
055  protected Context newContext(final Name prefix) throws NamingException {
056    return new MapContext(new HashMap<>(), this.environment, this.getNameParser(EMPTY_NAME), Objects.requireNonNull(prefix));
057  }
058
059  @Override
060  protected Context copy() throws NamingException {
061    return new MapContext(this.map, this.environment, this.getNameParser(EMPTY_NAME), this.prefix);
062  }
063
064  @Override
065  protected final boolean containsKey(final String mapKey) throws NamingException {
066    return this.map.containsKey(mapKey);
067  }
068  
069  @Override
070  protected final Object get(final String mapKey) throws NamingException {
071    return this.map.get(Objects.requireNonNull(mapKey));
072  }
073
074  @Override
075  protected final Set<String> keySet() {
076    return this.map.keySet();
077  }
078  
079  @Override
080  protected final Object remove(final String key) throws NamingException {
081    return this.map.remove(Objects.requireNonNull(key));
082  }
083
084  @Override
085  protected final Object put(final String key, final Object value) throws NamingException {
086    return this.map.put(Objects.requireNonNull(key), Objects.requireNonNull(value));
087  }
088
089  @Override
090  protected final String extractKey(final Name name) throws NamingException {
091    final Name compoundName = this.toCompoundName(Objects.requireNonNull(name));
092    assert compoundName != null;
093    final String returnValue;
094    if (compoundName.isEmpty()) {
095      returnValue = "";
096    } else {
097      returnValue = compoundName.get(0);
098    }
099    return returnValue;
100  }
101
102  @Override
103  public void rename(final Name oldName, final Name newName) throws NamingException {
104    Objects.requireNonNull(oldName);
105    Objects.requireNonNull(newName);
106    if (oldName.isEmpty()) {
107      throw new InvalidNameException("oldName.isEmpty()");
108    } else if (newName.isEmpty()) {
109      throw new InvalidNameException("newName.isEmpty()");
110    }
111
112    final String newKey = this.toCompoundName(newName).toString();
113    if (this.map.containsKey(newKey)) {
114      throw new NameAlreadyBoundException(newName.toString());
115    }
116
117    final String oldKey = this.toCompoundName(oldName).toString();
118    if (!this.map.containsKey(oldKey)) {
119      throw new NameNotFoundException(oldName.toString());
120    }
121    
122    this.put(newKey, this.remove(oldKey));
123  }
124
125}
